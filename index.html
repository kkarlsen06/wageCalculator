function calculateWage(shift) {
  let basePay = shift.hours * hourlyRate;
  let bonus = 0;

  // Helgetillegg
  if (isWeekend(shift.date)) {
    bonus += basePay * 0.18;
  }

  return basePay + bonus;
}

Om meg

Jeg er en lidenskapelig utvikler som spesialiserer meg på moderne webutvikling med fokus på brukervennlighet og ytelse. Lønnskalkulatoren er et resultat av min erfaring med komplekse beregningssystemer og ønsket om å forenkle hverdagen for folk med varierende arbeidstid.

Med bakgrunn innen både frontend og backend utvikling, lager jeg løsninger som er både teknisk solide og intuitive å bruke.
